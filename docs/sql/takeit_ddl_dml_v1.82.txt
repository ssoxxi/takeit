-- 상점구역
CREATE TABLE SHOP_LOC (
	SHOP_LOC_CODE VARCHAR2(10) NOT NULL -- 상점구역코드,
  ,	SHOP_LOC_NAME VARCHAR2(50) NOT NULL -- 상점구역이름
  ,	SHOP_LOC_LAT VARCHAR2(20) NOT NULL  -- 위도
  ,	SHOP_LOC_LNG VARCHAR2(20) NOT NULL -- 경도
  
  ,CONSTRAINT PK_SHOP_LOC PRIMARY KEY(SHOP_LOC_CODE)
);
-- 상점카테고리
CREATE TABLE SHOP_CATEGORY (
	SHOP_CATEGORY_NO VARCHAR2(10) NOT NULL -- 카테고리번호,
  ,	SHOP_CATEGORY VARCHAR2(20) NOT NULL -- 카테고리이름
  
  , CONSTRAINT PK_SHOP_CATEGORY PRIMARY KEY(SHOP_CATEGORY_NO)
);
-- 회원구역
CREATE TABLE MEMBER_LOC (
	MEMBER_LOC_NO VARCHAR2(10) NOT NULL -- 회원구역번호,
  ,	SHOP_LOC_CODE VARCHAR2(10) NOT NULL -- 상점구역코드,
  ,	MEMBER_LOC_NAME VARCHAR2(50) NOT NULL -- 회원구역이름
  
  , CONSTRAINT PK_MEMBER_LOC PRIMARY KEY(MEMBER_LOC_NO, SHOP_LOC_CODE)
  , CONSTRAINT FK_SHOP_LOC_TO_MEMB_LOC FOREIGN KEY (SHOP_LOC_CODE) REFERENCES SHOP_LOC(SHOP_LOC_CODE) ON DELETE CASCADE
);
-- 판매자
CREATE TABLE SELLER (
	SELLER_ID varchar2(50) NOT NULL -- 판매자회원아이디,
  ,	SELLER_PW VARCHAR2(50) NOT NULL -- 비밀번호,
  ,	NAME VARCHAR(30) NOT NULL -- 이름,
  ,	MOBILE CHAR(13) NOT NULL -- 휴대폰,
  ,	EMAIL VARCHAR(100) NOT NULL -- 이메일,
  ,	ENTRY_DATE TIMESTAMP DEFAULT SYSDATE NOT NULL -- 가입일자,
  ,	POSTNO CHAR(5) NOT NULL -- 우편번호,
  ,	ADDRESS VARCHAR(255) NOT NULL -- 도로명주소,
  ,	ADDRESS_DETAIL VARCHAR(255) NOT NULL -- 상세주소,
  ,	GRADE CHAR(1) DEFAULT 'S' NULL -- 등급,
  ,	SELLER_NO CHAR(12) UNIQUE NOT NULL -- 사업자등록번호,
  ,	SHOP_MOBILE VARCHAR2(13) NOT NULL -- 상점연락처,
  ,	SHOP_NAME VARCHAR2(50) NOT NULL -- 상점명,
  ,	CUST_SCORE NUMBER(3,1) DEFAULT 0.0 NOT NULL -- 고객평점,
  ,	SHOP_KAKAO_ID VARCHAR2(50) NULL -- 카카오톡아이디,
  ,	SHOP_IMG VARCHAR2(200) NULL -- 상점이미지,
  ,	SHOP_CATEGORY_NO VARCHAR2(10) NOT NULL -- 카테고리번호,
  ,	SHOP_LOC_CODE VARCHAR2(10) NOT NULL -- 상점구역코드
  
  , CONSTRAINT PK_SELLER_MEMBER PRIMARY KEY(SELLER_ID)
  , CONSTRAINT FK_SHOP_CATE_TO_SEL_MEMB FOREIGN KEY(SHOP_CATEGORY_NO) REFERENCES SHOP_CATEGORY(SHOP_CATEGORY_NO) ON DELETE CASCADE
  , CONSTRAINT FK_SHOPLOC_TO_SELMEMB FOREIGN KEY(SHOP_LOC_CODE) REFERENCES SHOP_LOC(SHOP_LOC_CODE) ON DELETE CASCADE
);

-- 일반

-- 상품카테고리
CREATE TABLE ITEM_CATEGORY (
	ITEM_CATEGORY_NO VARCHAR2(10) NOT NULL -- 카테고리번호,
  ,	ITEM_CATEGORY_NAME VARCHAR2(20) NOT NULL -- 카테고리이름,
  ,	EXPIRATION_DATE VARCHAR2(20) -- 유통기한,
  ,	NOTICE VARCHAR2(500) NOT NULL -- 안내사항,
  ,	FRESH_PERCENT NUMBER(4,1) NULL -- 신선도,
  ,	PACK_TYPE_NO VARCHAR2(5) NOT NULL -- 포장타입번호
  
  , CONSTRAINT PK_ITEM_CATEGORY PRIMARY KEY(ITEM_CATEGORY_NO)

-- 상품
CREATE TABLE ITEM (
	ITEM_NO VARCHAR2(20) NOT NULL -- 상품번호,
  ,	SELLER_ID varchar2(50) NOT NULL -- 판매자회원아이디,
  ,	ITEM_NAME VARCHAR2(50) NOT NULL -- 상품명,
  ,	ITEM_PRICE NUMBER(8) NOT NULL -- 판매가,
  ,	SALES_UNIT VARCHAR2(20) NULL -- 판매단위,
  ,	ITEM_ORIGIN VARCHAR2(20) NOT NULL -- 원산지,
  ,	ITEM_STOCK NUMBER(3,0) NOT NULL -- 재고량,
  ,	ITEM_IMG VARCHAR2(200) NOT NULL -- 상품이미지,
  ,	ITEM_CUST_SCORE NUMBER(3,1) NOT NULL -- 고객평점,
  ,	ITEM_INPUT_DATE TIMESTAMP DEFAULT SYSDATE NOT NULL -- 등록일자,
  ,	DISC_RATE NUMBER(3,0) NULL -- 할인율,
  ,	ITEM_TAKEIT CHAR(1) DEFAULT 'F' NOT NULL -- 잇거래여부,
  ,	ITEM_CATEGORY_NO VARCHAR2(10) NOT NULL -- 카테고리번호
  
  , CONSTRAINT PK_ITEM PRIMARY KEY(ITEM_NO)
  , CONSTRAINT FK_ITEM_CATE_TO_ITEM FOREIGN KEY(ITEM_CATEGORY_NO) REFERENCES ITEM_CATEGORY(ITEM_CATEGORY_NO) ON DELETE CASCADE
  , CONSTRAINT FK_SEL_MEMB_TO_ITEM FOREIGN KEY(SELLER_ID) REFERENCES SELLER(SELLER_ID) ON DELETE CASCADE
);

-- 배송상태
CREATE TABLE SHIPPING (
	SHIP_STATUS_CODE VARCHAR2(10) NOT NULL -- 배송상태코드,
  ,	SHIP_STATUS VARCHAR2(20) NOT NULL -- 배송상태이름
  
  , CONSTRAINT PK_SHIPPING PRIMARY KEY(SHIP_STATUS_CODE)
);
-- 주문
CREATE TABLE ORDERS (
	ORDER_NO VARCHAR2(20) NOT NULL -- 주문번호,
  ,	RECEIVE_METHOD VARCHAR2(20) NOT NULL -- 수령방법,
  ,	RECIPIENT_NAME VARCHAR2(50) NOT NULL -- 수령인,
  ,	RECIPIENT_POSTNO CHAR(5) NULL -- 수령인우편번호,
  ,	RECIPIENT_ADDR VARCHAR2(500) NULL -- 수령인 상세주소,
  ,	RECIPIENT_MOBILE CHAR(13) NOT NULL -- 수령인 연락처,
  ,	SHIP_REQUEST VARCHAR2(100) NULL -- 배송요청사항,
  , ORDER_PRICE NUMBER(10) NOT NULL -- 총주문금액
  , ORDER_CANCEL_REQ CHAR(1) DEFAULT 'F' -- 주문취소요청
  , ORDER_CANCEL CHAR(1) DEFAULT 'F' --주문취소
  ,	SHIP_STATUS_CODE VARCHAR2(10) NULL -- 배송상태코드,
  ,	MEMBER_ID varchar2(50) NULL -- 일반회원아이디

  
  , CONSTRAINT PK_ORDERS PRIMARY KEY(ORDER_NO)
  , CONSTRAINT FK_SHIP_TO_ORDERS FOREIGN KEY(SHIP_STATUS_CODE) REFERENCES SHIPPING(SHIP_STATUS_CODE) ON DELETE CASCADE
  , CONSTRAINT FK_GEN_MEMB_TO_ORDERS FOREIGN KEY(MEMBER_ID) REFERENCES MEMBER(MEMBER_ID) ON DELETE CASCADE
);
-- 주문상품상세
CREATE TABLE ORDER_DETAIL (
	ITEM_NO VARCHAR2(20) NOT NULL -- 상품번호,
  ,	ORDER_NO VARCHAR2(20) NOT NULL -- 주문번호,
  ,	ITEM_QTY NUMBER(3) NOT NULL -- 상품주문수량
  ,	ITEM_PAY_PRICE NUMBER(10) NOT NULL
  
  , CONSTRAINT PK_ORDER_DETAIL PRIMARY KEY(ITEM_NO, ORDER_NO)
  , CONSTRAINT FK_ITEM_TO_ORDER_DETAIL FOREIGN KEY(ITEM_NO) REFERENCES ITEM(ITEM_NO) ON DELETE CASCADE
  , CONSTRAINT FK_ORDERS_TO_ORDER_DETAIL FOREIGN KEY(ORDER_NO) REFERENCES ORDERS(ORDER_NO) ON DELETE CASCADE
);

-- 잇거래
CREATE TABLE TAKEIT (
	TAKEIT_NO VARCHAR2(20) NOT NULL -- 잇거래번호,
  ,	TAKEIT_PRICE NUMBER(10,0) NOT NULL -- 모집금액,
  , TAKEIT_CURR_PRICE NUMBER(10,0) DEFAULT 0 NOT NULL -- 현재 모집된 금액
  ,	TAKEIT_DATE TIMESTAMP DEFAULT SYSDATE NOT NULL -- 모집시작일자,
  ,	TAKEIT_CUST_SCORE NUMBER(3,1) DEFAULT 5.0 NOT NULL -- 고객평점,
  ,	TAKEIT_ALIVE CHAR(1) DEFAULT 'T' NOT NULL -- 잇거래진행여부,
  ,	MEMBER_LOC_NO VARCHAR2(10) NOT NULL -- 회원구역번호,
  ,	SHOP_LOC_CODE VARCHAR2(10) NOT NULL -- 상점구역코드
 
 
  , CONSTRAINT PK_TAKEIT PRIMARY KEY(TAKEIT_NO)
  , CONSTRAINT FK_MEMB_LOC_TO_TAKEIT FOREIGN KEY(MEMBER_LOC_NO, SHOP_LOC_CODE) REFERENCES MEMBER_LOC(MEMBER_LOC_NO, SHOP_LOC_CODE) ON DELETE CASCADE
);

-- 잇거래주문목록
CREATE TABLE TAKEIT_DETAIL (
	TAKEIT_NO VARCHAR2(20) NOT NULL -- 잇거래번호,
  ,	ORDER_NO VARCHAR2(20) NOT NULL -- 주문번호,
  
  , CONSTRAINT PK_TAKEIT_DETAIL PRIMARY KEY(TAKEIT_NO, ORDER_NO)
  , CONSTRAINT FK_ORDERS_TO_TAKEIT_D FOREIGN KEY(ORDER_NO) REFERENCES ORDERS(ORDER_NO) ON DELETE CASCADE
  , CONSTRAINT FK_TAKEIT_TO_TAKEIT_D FOREIGN KEY(TAKEIT_NO) REFERENCES TAKEIT(TAKEIT_NO) ON DELETE CASCADE
);

-- 게시글
CREATE TABLE BOARD (
	BOARD_NO VARCHAR2(20) NOT NULL -- 게시글번호,
  ,	BOARD_WRITER varchar2(50) NOT NULL -- 작성자아이디
  ,	BOARD_TITLE VARCHAR2(100) NOT NULL -- 게시글 제목,
  ,	BOARD_CONTENTS VARCHAR2(1000) NOT NULL -- 게시글 내용,
  ,	BOARD_VIEWS NUMBER(10)DEFAULT 0 NULL -- 게시글 조회수,
  ,	BOARD_PICKS NUMBER(10) DEFAULT 0 NULL -- 게시글 추천수,
  ,	BOARD_DATE TIMESTAMP DEFAULT SYSDATE NOT NULL -- 게시글 작성일자,
  ,	BOARD_CATEGORY_NO VARCHAR2(10) NOT NULL -- 게시글 카테고리 번호,
  ,	ITEM_NO VARCHAR2(20) NULL -- 상품번호,

  , CONSTRAINT PK_BOARD PRIMARY KEY(BOARD_NO)
  , CONSTRAINT FK_BOARD_CATE_TO_BOARD FOREIGN KEY(BOARD_CATEGORY_NO) REFERENCES BOARD_CATEGORY(BOARD_CATEGORY_NO) ON DELETE CASCADE
  , CONSTRAINT FK_ITEM_TO_BOARD FOREIGN KEY(ITEM_NO) REFERENCES ITEM(ITEM_NO) ON DELETE CASCADE
);

    
-- 후기
CREATE TABLE REVIEW (
    REVIEW_NO VARCHAR2(20) NOT NULL -- 후기 번호
  , MEMBER_ID VARCHAR2(50) NOT NULL --일반회원아이디
  , ITEM_NO VARCHAR2(50) NOT NULL -- 상품번호
  , REVIEW_DATE TIMESTAMP DEFAULT SYSDATE NOT NULL -- 후기 등록일자
  , REVIEW_TITLE VARCHAR2(500) NOT NULL --후기제목
  , REVIEW_CONTENTS VARCHAR2(2000) NOT NULL --후기내용
  , REVIEW_VIEWS NUMBER(10)DEFAULT 0 NULL -- 후기 조회수,
  , REVIEW_SCORE NUMBER(3,1) DEFAULT 0.0 NOT NULL -- 후기 평점
  , REVIEW_IMG VARCHAR2(200) NULL -- 후기사진
  
  , CONSTRAINT PK_REVIEW PRIMARY KEY(REVIEW_NO)
  , CONSTRAINT FK_REVIEW_MEMB_ID FOREIGN KEY(MEMBER_ID) REFERENCES MEMBER(MEMBER_ID) ON DELETE CASCADE
  , CONSTRAINT FK_REVIEW_ITEM_NO FOREIGN KEY(ITEM_NO) REFERENCES ITEM(ITEM_NO)  ON DELETE CASCADE
);
